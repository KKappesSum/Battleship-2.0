objo := Executive.o Admiral.o Ship.o Grid.o
objc := $(objo:.o=.cpp)
flags := -std=c++11 -g -Wall

all: main.o $(objo)
	g++ $(flags) main.o $(objo) -o Battleship

$(objo): $(objo:.o=.h) $(objc)
	g++ $(flags) -c $(objc)

main.o: main.cpp
	g++ $(flags) -c main.cpp

//WINDOWS
clean:
	del -f main.o $(objo) Battleship.exe

//UNIX
clean:
	rm -f main.o $(objo) Battleship
